### Opcode 核心格式

```
[封包頭部 (4 bytes)][Opcode (2 bytes)][資料...]

Opcode 格式：2 bytes (little endian)
例如：Opcode 0x0001 在封包中顯示為 01 00
```

### MapleCrypto 頭部混淆機制

```
原始 opcode: 0x0001 (例如 LOGIN_PASSWORD)
       IV: [0x7A, 0x52, ...]

Step 1: XOR + SHIFT
  verify = (opcode ^ iv[0]) + (iv[1] << 8)
        = (0x0001 ^ 0x7A) + (0x52 << 8)
        = 0x007B + 0x5200
        = 0x527B

Step 2: 位旋轉 (rotate left 3 bits)
  verify = (verify << 3) | (verify >> 13)
        = 0x527B << 3 | 0x527B >> 13
        = 0x29BD8 | 0x00
        = 0x29BD8

Step 3: 取低位 16 bits
  verify = 0x29BD8 & 0xFFFF = 0x9BD8
```

### 動態配方系統腳本源（無需 SQL）

```javascript
/*
 * 動態配方系統 - 玩家提交材料換取成品
 * NPC ID: 9010010
 */
var status = 0;

var recipes = Array(
    {name: "楓之葉地圖", materials: [[4000000, 10], [4000001, 5]], result: [[2300000, 1]], meso: 1000},
    {name: "傳說武器", materials: [[4003000, 100], [4003001, 50]], result: [[1302000, 1]], meso: 50000}
);

function start() { status = -1; action(1, 0, 0); }

function action(mode, type, selection) {
    if (mode == -1) { cm.dispose(); return; }
    if (mode == 0 && type == 1) { cm.dispose(); return; }
    if (mode == 1) status++; else status--;

    if (status == 0) {
        var msg = "歡迎來到工藝系統！\r\n請選擇配方：\r\n";
        for (var i = 0; i < recipes.length; i++) {
            msg += "#L" + i + "#" + recipes[i].name + "#l\r\n";
        }
        cm.sendNext(msg);
    } else if (status == 1) {
        var recipe = recipes[selection];
        var canCraft = true;
        var msg = "你需要以下材料：\r\n";
        for (var i = 0; i < recipe.materials.length; i++) {
            var mat = recipe.materials[i];
            var have = cm.getQuantityOfItem(mat[0]);
            var need = mat[1];
            msg += "#i" + mat[0] + "# #t" + mat[0] + "#: " + have + "/" + need + "\r\n";
            if (have < need) canCraft = false;
        }
        if (cm.getMesos() < recipe.meso) canCraft = false;
        msg += "加工費: " + recipe.meso + " 金幣\r\n";
        msg += canCraft ? "#g可以製作#k" : "#r材料不足#k";
        cm.sendYesNo(msg);
    } else if (status == 2) {
        var recipe = recipes[selection];
        for (var i = 0; i < recipe.materials.length; i++) {
            cm.gainItem(recipe.materials[i][0], -recipe.materials[i][1]);
        }
        cm.gainMesos(-recipe.meso);
        for (var i = 0; i < recipe.result.length; i++) {
            cm.gainItem(recipe.result[i][0], recipe.result[i][1]);
        }
        cm.sendOk("製作成功！");
        cm.dispose();
    }
}
```

### 完整架設流程
```
1. 安裝 JDK 8 + JCE 8
2. 安裝 MySQL Server 5.5
3. 下載 v83 localhost 客戶端
4. 架設 Java 伺服器（NetBeans / IntelliJ）
5. 執行 SQL 初始化資料庫
6. 設定 ServerConstants.java 中的 IP 和埠口
7. HxD 修改客戶端 IP（三個 127.0.0.1 全部替換）
8. 啟動客戶端和伺服器
```

### LAN 多人連線必檢查清單
- [ ] 確認兩台 PC 在同一 LAN 網段
- [ ] 確認伺服器 listen 0.0.0.0（而非 127.0.0.1）
- [ ] 確認防火牆開啟 7575 和 8484 埠口
- [ ] 若錯誤碼 0x80004005 → 防火牆/權限問題
- [ ] 若錯誤碼 -2147467259 → 同上
