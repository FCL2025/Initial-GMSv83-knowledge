### 2. 抓包分析

#### 🆕 MapleCrypto 完整加密流程（maplelib Go 語言實作）

**來源**：https://pkg.go.dev/github.com/defaultmagi/maplelib

**關鍵原理**：
> MapleStory 使用兩組金鑰：一組用於加密發送封包，另一組用於解密接收封包。兩組金鑰在客戶端連接時隨機生成，並在第一個未加密的 Hello Packet 中傳送。

**完整封包加密流程（v83）**：
```
┌─────────────────────────────────────────────────────┐
│ 發送端（客戶端或伺服器）                            │
│                                                     │
│ 1. 構造明文封包                                    │
│    [Opcode (2 bytes)] [Data ...]                   │
│                                                     │
│ 2. Shanda 對稱加密                                 │
│    - 使用版本特定的靜態金鑰陣列                     │
│    - 每 byte 與金鑰進行 XOR 運算                   │
│    - v83 仍使用，v118+ 已移除                      │
│                                                     │
│ 3. AES-OFB 加密（主要加密）                        │
│    - 256-bit key                                  │
│    - IV（初始化向量）在 Hello Packet 協商          │
│    - OFB (Output Feedback) 模式                   │
│                                                     │
│ 4. MapleCrypto 頭部混淆                            │
│    - 4 bytes header                                │
│    - 包含：混淆 opcode + 封包長度                  │
│    - 使用 IV 進行動態生成                          │
│                                                     │
│ 5. 網路傳輸                                        │
└─────────────────────────────────────────────────────┘
```

**AES-OFB 加密關鍵參數**：
| 參數 | 說明 |
|------|------|
| Key | 256-bit，版本決定的靜態密鑰 |
| IV | 4 bytes，在 Hello Packet 中動態協商 |
| Mode | OFB (Output Feedback) |
| Block Size | 16 bytes |

**Hello Packet 協商流程**：
```
1. 客戶端生成隨機 SendIV 和 RecvIV
2. 客戶端發送 Hello Packet（明文）：
   - 版本號
   - SendIV (4 bytes)
   - RecvIV (4 bytes)
3. 伺服器回應 Hello Packet（明文）
4. 雙方使用收到的 IV 初始化 AES-OFB
5. 之後所有封包均加密
```

**Go 語言 MapleCrypto 實現**：
```go
// MapleCrypto header 生成
func (a *AES) GetPacketHeader(opcode uint16, length int32) []byte {
    maskedVersion := ((a.version >> 8) & 0xFF) | ((a.version << 8) & 0xFF00)
    // 這解釋了為何版本號在封包中是 byte-swapped
    
    verify := (int(opcode) ^ int(a.iv&0xFF)) + int((a.iv>>8)&0xFF)*0x100
    verify = (verify << 3) | (verify >> 5)
    return []byte{
        byte(verify & 0xFF),
        byte((verify >> 8) & 0xFF),
        byte(length & 0xFF),
        byte((length >> 8) & 0xFF),
    }
}
```

**AES 版本位元組交換問題**：
```go
// 為何版本 83 (0x0053) 在封包中顯示為 0x5300？
const left  = (version >> 8) & 0xFF;  // 0x53 → 0x00
const right = (version << 8) & 0xFF00; // 0x53 → 0x5300
const maskedVersion = left | right; // = 0x5300
```
這就是為什麼在 Wireshark 中看到 `53 00` 而非 `00 53`。

#### 🆕 Shanda 加密實作（v83）

**來源**：https://github.com/Kevin-Jin/verticle-story/blob/master/src/main/java/net/pjtb/vs/playerside/ClientEncryption.java

**Java 實現關鍵片段**：
```java
public class ClientEncryption {
    private static final int[] SHANDA_KEY = {/* 版本特定 */};
    
    public void encrypt(byte[] data, int offset, int length) {
        int siv = this.siv; // 當前 IV
        for (int i = offset; i < offset + length; i++) {
            int a = (siv >> 24) & 0xFF;
            int b = (siv >> 16) & 0xFF;
            int c = (siv >> 8) & 0xFF;
            int d = siv & 0xFF;
            
            int e = (a ^ ((data[i] & 0xFF))) + (b * c | b) + d;
            data[i] = (byte) (e & 0xFF);
            
            // 更新 IV
            siv = (siv * 0xBABE + (e & 0xFFFF)) & 0xFFFFFFFF;
        }
        this.siv = siv;
    }
}
```

**問題1：無窮對話迴圈**

#### 🆕 NPC 腳本源：完整任務接收+交付腳本

```javascript
/*
 * 複合任務 NPC
 * NPC ID: 9000001
 * 任務 ID: 1000
 * 功能：接收任務 → 收集物品 → 領取獎勵
 */
var status = 0;
var questId = 1000;
var reqItem = 4000000;  // 黑木妖的葉子
var reqCount = 10;
var rewardItem = 1000000;  // 楓幣
var rewardMeso = 5000;

function start() {
    status = -1;
    action(1, 0, 0);
}

function action(mode, type, selection) {
    if (mode == -1) {
        cm.dispose();
        return;
    }
    
    if (mode == 0 && type == 1) {
        cm.dispose();
        return;
    }
    
    if (mode == 1)
        status++;
    else
        status--;
    
    var qs = cm.getQuestStatus(questId);
    
    if (status == 0) {
        if (qs == 0) {
            // 任務未接
            cm.sendNext("冒險家！我需要你的幫助。\r\n" +
                "最近 #r黑木妖#k 在村莊附近出沒，請幫我收集 #b" + reqCount + "個黑木妖的葉子#k。");
        } else if (qs == 1) {
            // 任務進行中
            var progress = cm.getQuestProgress(questId);
            cm.sendNext("你目前收集了 #b" + progress + "#k / " + reqCount + " 個葉子。\r\n" +
                "繼續加油！");
        } else {
            // 任務已完成
            cm.sendOk("你已經完成任務了，謝謝！");
            cm.dispose();
        }
    } else if (status == 1) {
        if (qs == 0) {
            // 開始任務
            cm.startQuest(questId);
            cm.sendOk("去吧！收集 #b" + reqCount + "個#k 葉子後回來找我。");
        } else if (qs == 1) {
            // 檢查物品
            var count = cm.getItemQuantity(reqItem);
            if (count >= reqCount) {
                // 完成任務
                cm.gainItem(reqItem, -reqCount);  // 移除物品
                cm.gainMeso(rewardMeso);
                cm.completeQuest(questId);
                cm.sendOk("太棒了！你幫了大忙！\r\n" +
                    "這是 #b" + rewardMeso + " 楓幣#k 作為獎勵。");
            } else {
                cm.sendOk("你還需要 #b" + (reqCount - count) + "個#k 葉子。");
            }
        }
        cm.dispose();
    }
}
```
